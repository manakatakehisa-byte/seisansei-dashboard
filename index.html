function doGet(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName('生産性（新）'); const data = sheet.getDataRange().getValues(); const result = []; for (let i = 2; i < data.length; i++) { const row = data[i]; if (!row[0] || typeof row[0] === 'string') continue; if (!row[2] || row[2] === '') continue; let timeHours = 0; if (row[5] instanceof Date) { timeHours = row[5].getHours() + row[5].getMinutes() / 60; } else if (typeof row[5] === 'number') { timeHours = row[5] * 24; } result.push({ date: Utilities.formatDate(new Date(row[0]), 'Asia/Tokyo', 'yyyy/MM/dd'), type: String(row[1] || ''), name: String(row[2] || ''), quantity: Number(row[3] || 0), note: String(row[4] || ''), time: timeHours }); } const output = ContentService.createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); return output; }
